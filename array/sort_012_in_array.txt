class Solution
{
    public:
    void sort012(int a[], int n)
    {
   int low=0;
   int high=n-1;
   int start=0;
   while(start<high)
   {
      if(a[start]==0)
      {
          swap(a[start],a[low]);
          start++;
          low++;
      }
      else if(a[start]==2)
      {
          swap(a[start],a[high]);
          high--;
      }
      else
      {
          start++;
      }
    }
}
};
