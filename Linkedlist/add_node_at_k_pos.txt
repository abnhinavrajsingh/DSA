#include<iostream>
#include<bits/stdc++.h>
using namespace std;

struct Node{
  int data;
  Node* next;
  
  Node(int data)
  {
      this->data = data;
      this->next = NULL;
  }
};

Node* insert_at_end(Node* &head,int nodeval)
{
   if(!head) return new Node(nodeval);
   else
   {   Node* temp = head;
       while(temp->next!=NULL)
       {
          temp = temp->next; 
       }
       Node* newnode = new Node(nodeval);
       temp->next = newnode;
       temp = head;
       return head;
   }
}

Node* insert_node_at_k_position(Node* &head,int nodeval,int pos)
{
    if(!head) return head;
    Node* temp = head;
    if(pos==1)
    {
    Node* newnode = new Node(nodeval);
    newnode->next = temp->next;
    temp = newnode;
    return temp;
    }
    
    pos = pos-2;  // let assume 1st node is at 1st position (means not zero)
    while(pos--)
    {
        temp = temp->next;    
    }
    Node* newnode = new Node(nodeval);
    newnode->next = temp->next;
    temp->next = newnode;
    return head;
}
void print(Node* head)
{
    while(head!=NULL)
    {
        cout<<head->data<<" ";
        head = head->next;
    }
}
int main()
{
    Node* head = NULL;
    int n;
    cin>>n;
    while(n--)
    {
        int nodeval;
        cin>>nodeval;
        head = insert_at_end(head,nodeval);
    }
    head = insert_node_at_k_position(head,1000,1);  //nodeval=1000 and k = 3
    print(head);
}
