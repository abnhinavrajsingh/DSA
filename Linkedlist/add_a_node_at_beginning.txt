#include<iostream>
#include<bits/stdc++.h>
using namespace std;

struct Node{
  int data;
  Node* next;
  
  Node(int data)
  {
      this->data = data;
      this->next = NULL;
  }
};

Node* insert(Node* head,int nodeval)
{
    if(head==NULL) return new Node(nodeval);
    else
    {
     Node* newnode = new Node(nodeval);
    newnode->next = head;
    head = newnode;
    return head;  
    }
    
}

void print(Node* head)
{
    while(head!=NULL)
    {
        cout<<head->data<<" ";
        head = head->next;
    }
}
int main()
{
    Node* head = NULL;
    int n;
    cin>>n;
    while(n--)
    {
        int nodeval;
        cin>>nodeval;
        head = insert(head,nodeval);
    }
    print(head);
}
