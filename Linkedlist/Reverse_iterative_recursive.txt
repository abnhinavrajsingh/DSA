#include<iostream>
#include<bits/stdc++.h>
using namespace std;

struct Node{
  int data;
  Node* next;
  
  Node(int data)
  {
      this->data = data;
      this->next = NULL;
  }
};

Node* insert(Node* head,int nodeval)
{
    if(head==NULL) return new Node(nodeval);
    else
    {
     Node* newnode = new Node(nodeval);
    newnode->next = head;
    head = newnode;
    return head;  
    }
    
}

Node* reverse_iterative(Node* head)
{
    Node* curr = head;
    Node *next =NULL,*prev = NULL;
    while(curr)
    {
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
}

Node* reverse_recursive(Node* head)
{
    if(head==NULL || head->next==NULL)
        return head;
    Node* temp_head = reverse_recursive(head->next);
    head->next->next = head;
    head->next =NULL;
    return temp_head;
}

void print(Node* head)
{
    while(head!=NULL)
    {
        cout<<head->data<<" ";
        head = head->next;
    }
}

int main()
{
    Node* head = NULL;
    int n;
    cin>>n;
    while(n--)
    {
        int nodeval;
        cin>>nodeval;
        head = insert(head,nodeval);
    }
    cout<<"linkedlist is : ";
    print(head);
    cout<<endl;
    
    head = reverse_iterative(head);
    cout<<"reversed linkedlist is (iterative): ";
    print(head);
    cout<<endl;
    
    head = reverse_recursive(head);
    cout<<"reversed linkedlist is (Recursive): ";
    print(head);
    cout<<endl;
}
