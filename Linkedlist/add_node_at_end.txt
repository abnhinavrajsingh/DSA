#include<iostream>
#include<bits/stdc++.h>
using namespace std;

struct Node{
  int data;
  Node* next;
  
  Node(int data)
  {
      this->data = data;
      this->next = NULL;
  }
};

Node* insert_at_end(Node* &head,int nodeval)
{
   if(!head) return new Node(nodeval);
   else
   {   Node* temp = head;
       while(temp->next!=NULL)
       {
          temp = temp->next; 
       }
       Node* newnode = new Node(nodeval);
       temp->next = newnode;
       temp = head;
       return head;
   }
}

void print(Node* head)
{
    while(head!=NULL)
    {
        cout<<head->data<<" ";
        head = head->next;
    }
}
int main()
{
    Node* head = NULL;
    int n;
    cin>>n;
    while(n--)
    {
        int nodeval;
        cin>>nodeval;
        head = insert_at_end(head,nodeval);
    }
    print(head);
}
