#include<bits/stdc++.h>
using namespace std;
void insertion_sort(int arr[],int n)
{
    for(int i=1;i<n;++i)
    {
        int j = i-1;
        int key = arr[i];
        while(j>=0 && arr[j]>key)
        {
            arr[j+1] = arr[j];
            j--;
        }
        arr[j+1] = key;
    }
}
int main()
{
    int arr[5]{3,2,4,1,5};
    int n = 5;
    insertion_sort(arr,n);
    for(int i=0;i<n;++i)
        cout<<arr[i]<<" ";
}
