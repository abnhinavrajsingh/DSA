#include<bits/stdc++.h>
using namespace std;

void merge(int arr[],int start,int mid,int end,int n)
{
    int start2 = mid+1;
    while(start<=mid && start2<=end)
    {
        if(arr[start]<arr[start2])
            start++;
        else
        {
            int index = start2;
            int value = arr[start2];
            while(index!=start)
            {
                arr[index] =arr[index-1];
                index--;
            }
               arr[start] = value;
               start++;
               mid++;
               start2++;
        }
    }
}

void merge_sort(int arr[],int start,int end,int n)
{
if(start<end)
{
    int mid = start+(end - start)/2;
    merge_sort(arr,start,mid,n);
    merge_sort(arr,mid+1,end,n);
    merge(arr,start,mid,end,n);
}
}

int main()
{
    int arr[5]{3,2,4,1,5};
    int n = 5;
    int start = 0;
    int end = n-1;
    merge_sort(arr,start,end,n);
    for(int i=0;i<n;++i)
        cout<<arr[i]<<" ";
}
