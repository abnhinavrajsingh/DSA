#include<bits/stdc++.h>
using namespace std;
void selection_sort(int arr[],int n)
{
    for(int i=0;i<n-1;++i)
    {
        int minindex = i;
        for(int j = i+1;j<n;++j)
        {
            if(arr[j]<arr[minindex])
                minindex = j;
        }
        swap(arr[minindex],arr[i]);
    }
}
int main()
{
    int arr[5]{3,2,4,1,5};
    int n = 5;
    selection_sort(arr,n);
    for(int i=0;i<n;++i)
        cout<<arr[i]<<" ";
}
