#include<bits/stdc++.h>
using namespace std;

int partition(int arr[],int start,int end)
{
   int pivot = arr[end];
   int i = start;
   int j = start;
   while(i<=end)
   {
       if(arr[i]>pivot)
            i++;
        else
        {
            swap(arr[i],arr[j]);
            i++;
            j++;
        }
   }
   return j-1;
}

void quick_sort(int arr[],int start,int end)
{
    if(start<end)
    {
        int pivot = partition(arr,start,end);
        quick_sort(arr,start,pivot-1);
        quick_sort(arr,pivot+1,end);
    }
}

int main()
{
    int arr[5]{3,2,4,1,5};
    int n = 5;
    int start = 0;
    int end = n-1;
    quick_sort(arr,start,end);
    for(int i=0;i<n;++i)
        cout<<arr[i]<<" ";
}
